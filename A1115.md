# A1115 Counting Nodes in a BST

## 1.题意理解
给出一个序列，要求逐个插入到空二叉搜索树中，然后计算最下面两层结点数之和。

## 2.思路分析
掌握最基本的操作，二叉树插入（注意题目中规定相等是左子树，右子树还是不插入），本题中规定小于等于都插在左子树。

然后是层序遍历，每层宽度的计算。

## 3.参考代码
```cpp
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <vector>
#include <queue>
using namespace std;

const int N = 1024;
int n, data[N];
vector<int> layer;

struct node
{
    int data;
    node *left, *right;
};

node *newNode(int v)
{
    node *nd = new node;
    nd->data = v;
    nd->left = nd->right = NULL;
    return nd;
}

void insert(node *&root, int x)
{
    if (root == NULL)
    {
        root = newNode(x);
        return;
    }
    if (x <= root->data)
        insert(root->left, x);
    else
        insert(root->right, x);
}

node *create(int data[], int n)
{
    node *root = NULL;
    for (int i = 0; i < n; i++)
        insert(root, data[i]);
    return root;
}

void bfs(node *root)
{
    queue<node *> q;
    node *last = root, *tail; // tail也可以不要，直接用q.back()获取
    int level = 0, cnt = 0;
    q.push(root);
    while (!q.empty())
    {
        node *now = q.front();
        cnt++;
        q.pop();

        if (now->left != NULL)
        {
            q.push(now->left);
            tail = now->left;
        }

        if (now->right != NULL)
        {
            q.push(now->right);
            tail = now->right;
        }

        if (last == now)
        {
            layer.push_back(cnt);
            level++;
            last = tail;
            cnt = 0;
        }
    }
}

int main()
{
    scanf("%d", &n);
    for (int i = 0; i < n; i++)
        scanf("%d", &data[i]);

    node *root = create(data, n);
    bfs(root);

    int n1 = layer[layer.size() - 1];
    int n2 = layer[layer.size() - 2];
    printf("%d + %d = %d\n", n1, n2, n1 + n2);

    return 0;
}
```
